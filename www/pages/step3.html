Step3: Zoom in to your location

You can zoom-in to the specified location with animation easily!

function onLocationSuccess(result) {
  map.animationCamera({
    //tilt: 10,
    //bearing: 230,
    target: result.latLng,
    zoom: 17
  }, onAnimationFinished);
}
function onAnimationFinished() {
  alert("Animation finished!");
}
