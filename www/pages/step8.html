Step8: Mapping image on the map

You can put your image on the map

var bounds = [
  new plugin.google.maps.LatLng(45.744757, 147.12085),
  new plugin.google.maps.LatLng(41.144941, 138.060425)
];
map.addGroundOverlay({
  'url': "www/images/hokkaido.png",
  'bounds': bounds,
  'opacity': 0.5
}, onGroundOverlayAdded);

function onGroundOverlayAdded(groundOverlay) {
  map.animateCamera({
    target: bounds
  }, onCameraUpdated);
}
